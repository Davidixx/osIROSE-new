686  if (rhsMemberItr == rhs.MemberEnd() || lhsMemberItr->value != rhsMemberItr->value)
687  return false;

692  if (data_.a.size != rhs.data_.a.size)
693  return false;
694  for (SizeType i = 0; i < data_.a.size; i++)

1517  // (excluding the terminating zero) and store a value to determine the length of the contained
1518  // string in the last character str[LenPos] by storing "MaxSize - length" there. If the string
1519  // to store has the maximal length of MaxSize then str[LenPos] will be 0 and therefore act as
1520  // the string terminator as well. For getting the string length back from that value just use
1521  // "MaxSize - str[LenPos]".

1523  // inline (for `UTF8`-encoded strings).

1530  inline SizeType GetLength() const { return (SizeType)(MaxSize - str[LenPos]); }
